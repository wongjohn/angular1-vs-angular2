Destructuring


'use strict';

function foo() {
    return [1,2,3];
}

    return {
        x: 4, y: 5, z:6
    };
}
var [a,b,c]= foo();
var {x:x,y:y,z:z} = bar();
console.log( a, b, c ); // 1 2 3
console.log( x, y, z ); // 4 5 6

function foo({x,y}){ console.log( x, y );
}
foo( { y: 1, x: 2 } );// 2 1
foo( { y: 42 } );// undefined 42
foo( {} );// undefined undefined

for..of Loops


'use strict';
var a = ["a","b","c","d","e"];
for(var idx in a){
    console.log( idx );// 0 1 2 3 4
}

for(var val of a){
    console.log( val );// "a" "b" "c" "d" "e"
}
//for..in loops over the keys/indexes in the a array, while for..of loops over the values in a.

Classes


'use strict';
class Foo {
    constructor(a,b) {
        this.x = a;
        this.y = b;
    }
    gimmeXY() {
        return this.x * this.y;
    }
}
class Bar extends Foo {
    constructor(a,b,c) {
        super( a, b );
        this.z = c;
    }
    gimmeXYZ() {
        return super.gimmeXY() * this.z;
    }
}
var b= new Bar(5,15,25);
b.x; // 5
b.y; // 15
b.z; // 25
b.gimmeXYZ(); // 1875
